#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>
#define MAX_LEN 10000
int main() {
    char str1[MAX_LEN + 2];
    char str2[MAX_LEN + 2];
    char result[MAX_LEN + 1];

    fgets(str1, sizeof(str1), stdin);
    fgets(str2, sizeof(str2), stdin);

    // Remove newline chars from input
    str1[strcspn(str1, "\r\n")] = 0;
    str2[strcspn(str2, "\r\n")] = 0;

    int len = strlen(str1);

    for (int i = 0; i < len; i++) {
        result[i] = (str1[i] == str2[i]) ? '0' : '1';
    }
    result[len] = '\0';

    printf("%s ", result);
    return 0;
}
